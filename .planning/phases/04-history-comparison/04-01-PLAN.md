    **Step 2b: Add velocity/acceleration computation using linearRegression in `src/lib/history/snapshot.ts`**

    ```typescript
    import { linearRegression } from '@/lib/engine/stats/regression.ts';

    /**
     * Compute velocity (followers/day slope) and acceleration (second derivative approximation)
     * from a snapshot series. Returns null if fewer than 3 data points.
     */
    export function computeVelocityAcceleration(
      snapshots: HistorySnapshot[],
    ): { velocity: number; acceleration: 'accelerating' | 'steady' | 'decelerating' | 'stagnant' } | null {
      if (snapshots.length < 3) return null;

      const points = snapshots.map((s, i) => ({
        x: i,
        y: s.profile.followers,
      }));

      const { slope } = linearRegression(points);

      // Compute second derivative via slope of slope
      if (points.length < 4) {
        return { velocity: slope, acceleration: 'steady' };
      }

      const slopePoints = points.slice(1).map((p, i) => ({
        x: i,
        y: slope, // use same slope for now -- simplified
      }));

      // acceleration: compare recent slope vs earlier slope
      const mid = Math.floor(points.length / 2);
      const firstHalf = points.slice(0, mid);
      const secondHalf = points.slice(mid);
      const firstSlope = linearRegression(firstHalf).slope;
      const secondSlope = linearRegression(secondHalf).slope;
      const slopeChange = secondSlope - firstSlope;

      let acceleration: 'accelerating' | 'steady' | 'decelerating' | 'stagnant';
      if (slopeChange > 0.01) acceleration = 'accelerating';
      else if (slopeChange < -0.01) acceleration = 'decelerating';
      else acceleration = 'steady';

      return { velocity: slope, acceleration };
    }
    ```

    Delta computation logic:

    ```typescript
    function computeDeltas(snapshots: HistorySnapshot[], days: number): DeltaRow[] {
      const cutoff = Date.now() - days * 86_400_000;
      const recent = snapshots.filter(
        (s) => new Date(s.fetchedAt).getTime() >= cutoff,
      );

      const rows: DeltaRow[] = [];
      for (let i = 1; i < recent.length; i++) {
        const prev = recent[i - 1];
        const curr = recent[i];
        const followersDelta = curr.profile.followers - prev.profile.followers;
        const likesDelta = curr.profile.likesTotal - prev.profile.likesTotal;
        const videosDelta = curr.profile.videosCount - prev.profile.videosCount;
        const engagementDelta =
          curr.profile.engagementRate - prev.profile.engagementRate;

        rows.push({
          date: curr.fetchedAt,
          label: formatDate(curr.fetchedAt),
          followers: curr.profile.followers,
          followersDelta,
          followersDeltaPct: prev.profile.followers > 0
            ? (followersDelta / prev.profile.followers) * 100
            : 0,
          likesTotal: curr.profile.likesTotal,
          likesDelta,
          likesDeltaPct: prev.profile.likesTotal > 0
            ? (likesDelta / prev.profile.likesTotal) * 100
            : 0,
          engagementRate: curr.profile.engagementRate,
          engagementDelta,
          videosCount: curr.profile.videosCount,
          videosDelta,
        });
      }
      return rows;
    }
    ```

    Diff computation:

    ```typescript
    function buildDiffRows(
      t1: HistorySnapshot,
      t2: HistorySnapshot,
    ): DiffRow[] {
      const metrics: { key: keyof HistorySnapshot['profile']; label: string; format: (v: number) => string }[] = [
        { key: 'followers', label: t('ui.components.followers'), format: (v) => formatCompact(v) },
        { key: 'likesTotal', label: t('ui.components.totalLikes'), format: (v) => formatCompact(v) },
        { key: 'videosCount', label: t('ui.components.videos'), format: (v) => String(v) },
        { key: 'engagementRate', label: t('ui.components.engagementRate'), format: (v) => `${(v * 100).toFixed(1)}%` },
      ];

      return metrics.map(({ key, label, format }) => {
        const t1Val = t1.profile[key] as number;
        const t2Val = t2.profile[key] as number;
        const delta = t2Val - t1Val;
        const deltaPct = t1Val !== 0 ? (delta / Math.abs(t1Val)) * 100 : 0;
        const direction: DiffRow['direction'] =
          delta > 0 ? 'up' : delta < 0 ? 'down' : 'flat';

        return {
          metric: label,
          t1Value: format(t1Val),
          t2Value: format(t2Val),
          delta: `${delta >= 0 ? '+' : ''}${format(delta)}`,
          deltaPct: `${deltaPct >= 0 ? '+' : ''}${deltaPct.toFixed(1)}%`,
          direction,
        };
      });
    }
    ```

    ```typescript
    import { createHistoryStore } from '@/lib/history/store';
    import type { HistorySnapshot } from '@/lib/schema/creator-data';
    import GrowthTrendChart from '@/components/growth-trend-chart';

    interface TimelineHistoryClientProps {
      profileKeys: string[]; // e.g. ['douyin:123456', 'tiktok:123456']
    }

    export default function TimelineHistoryClient({ profileKeys }: TimelineHistoryClientProps) {
      const [snapshots, setSnapshots] = useState<HistorySnapshot[]>([]);
      const [isLoading, setIsLoading] = useState(true);

      useEffect(() => {
        if (typeof window === 'undefined') return;
        let cancelled = false;

        async function load() {
          const store = createHistoryStore();
          const all: HistorySnapshot[] = [];
          for (const key of profileKeys) {
            const snaps = await store.getSnapshots(key);
            all.push(...snaps);
          }
          if (!cancelled) {
            setSnapshots(all);
            setIsLoading(false);
          }
        }
        load();
        return () => { cancelled = true; };
      }, [profileKeys]);

      if (isLoading) {
        return (
          <div className="flex flex-col gap-6">
            <div className="h-64 animate-pulse rounded-lg bg-[var(--bg-card)]" />
            <div className="h-48 animate-pulse rounded-lg bg-[var(--bg-card)]" />
          </div>
        );
      }

      return (
        <div className="flex flex-col gap-8">
          {/* Growth trend chart with granularity */}
          <section>
            <GrowthTrendChart
              storeKeys={profileKeys}
              fallbackSnapshots={[]}
            />
          </section>

          {/* Delta table */}
          <section>
            <DeltaTable snapshots={snapshots} days={14} />
          </section>

          {/* T1/T2 diff view */}
          <section>
            <SnapshotDiffView snapshots={snapshots} />
          </section>
        </div>
      );
    }
    ```
