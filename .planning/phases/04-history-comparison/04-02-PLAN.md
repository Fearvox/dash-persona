    ```typescript
    /**
     * Compare 2-5 arbitrary CreatorProfiles against each other.
     * Normalizes each metric to 0-100 percentile within the compared set
     * using empiricalPercentile, and assigns per-metric ranks.
     *
     * @param profiles Array of 2-5 CreatorProfile objects to compare.
     * @returns MultiCreatorComparison with normalized scores and per-metric rankings.
     */
    export function compareMultiCreator(
      profiles: CreatorProfile[],
    ): MultiCreatorComparison {
      if (profiles.length < 2 || profiles.length > 5) {
        return { summaries: [], metrics: [], bestPerMetric: {} };
      }

      const summaries = profiles.map(buildCreatorSummary);

      // Define metrics to compare
      const metricDefinitions: {
        key: keyof CreatorSummary;
        label: string;
        higherIsBetter: boolean;
      }[] = [
        { key: 'followers', label: t('ui.compare.followers'), higherIsBetter: true },
        { key: 'overallEngagementRate', label: t('ui.compare.engagementRate'), higherIsBetter: true },
        { key: 'postCount', label: t('ui.compare.postCount'), higherIsBetter: true },
        { key: 'totalViews', label: t('ui.compare.totalViews'), higherIsBetter: true },
        { key: 'totalEngagement', label: t('ui.compare.totalEngagement'), higherIsBetter: true },
      ];

      const metrics: MetricComparison[] = [];
      const bestPerMetric: Record<string, string> = {};

      for (const def of metricDefinitions) {
        const values = summaries.map((s) => ({
          id: s.id,
          label: s.label,
          raw: s[def.key] as number,
          normalized: 0,
          rank: 0,
        }));

        // Collect all values for percentile normalization
        const allValues = values.map((v) => v.raw).sort((a, b) => a - b);

        // Normalize each value to 0-100 percentile
        for (const v of values) {
          v.normalized = empiricalPercentile(v.raw, allValues);
        }

        // Sort by normalized value (desc if higherIsBetter, asc otherwise)
        values.sort((a, b) =>
          def.higherIsBetter
            ? b.normalized - a.normalized
            : a.normalized - b.normalized,
        );

        // Assign ranks
        values.forEach((v, i) => {
          v.rank = i + 1;
        });

        // Best is rank 1
        if (values.length > 0) {
          bestPerMetric[def.key] = values[0].id;
        }

        metrics.push({
          metric: def.label,
          values,
        });
      }

      return { summaries, metrics, bestPerMetric };
    }

    /**
     * Memoized version of compareMultiCreator.
     */
    export const compareMultiCreatorCached = memoize(compareMultiCreator);
    ```

    ```typescript
    export function CreatorPicker({
      selected,
      onChange,
      max = MAX_COMPARISON,
      min = MIN_COMPARISON,
    }: CreatorPickerProps) {
      const [allProfiles, setAllProfiles] = useState<CreatorProfile[]>([]);
      const [isLoading, setIsLoading] = useState(true);
      const [search, setSearch] = useState('');

      useEffect(() => {
        loadProfiles().then((profiles) => {
          const list = Object.values(profiles);
          setAllProfiles(list);
          setIsLoading(false);
        });
      }, []);

      const toggleCreator = useCallback(
        (profile: CreatorProfile) => {
          const isSelected = selected.some(
            (p) => p.platform === profile.platform && p.profileUrl === profile.profileUrl,
          );
          if (isSelected) {
            onChange(selected.filter((p) => p.profileUrl !== profile.profileUrl));
          } else if (selected.length < max) {
            onChange([...selected, profile]);
          }
        },
        [selected, onChange, max],
      );

      const filtered = allProfiles.filter((p) => {
        if (!search) return true;
        const q = search.toLowerCase();
        return (
          p.platform.toLowerCase().includes(q) ||
          p.profileUrl.toLowerCase().includes(q)
        );
      });

      if (isLoading) {
        return (
          <div className="flex flex-col gap-2">
            {[1, 2, 3].map((i) => (
              <div key={i} className="h-12 animate-pulse rounded bg-[var(--bg-card)]" />
            ))}
          </div>
        );
      }

      if (allProfiles.length === 0) {
        return (
          <div className="rounded-lg border border-dashed border-[var(--border-subtle)] p-6 text-center">
            <p className="text-sm text-[var(--text-subtle)]">
              {t('ui.compare.noSavedProfiles')}
            </p>
            <p className="mt-1 text-xs text-[var(--text-subtle)]">
              {t('ui.compare.collectToCompare')}
            </p>
          </div>
        );
      }
    ```

    ```typescript
    const COMPARISON_STORAGE_KEY = 'dashpersona-comparison-set';

    export interface ComparisonSet {
      creatorIds: string[]; // format: "platform:uniqueId"
    }

    export function getComparisonSet(): ComparisonSet {
      if (typeof window === 'undefined') return { creatorIds: [] };
      try {
        const raw = sessionStorage.getItem(COMPARISON_STORAGE_KEY);
        if (!raw) return { creatorIds: [] };
        return JSON.parse(raw) as ComparisonSet;
      } catch {
        return { creatorIds: [] };
      }
    }

    export function setComparisonSet(set: ComparisonSet): void {
      if (typeof window === 'undefined') return;
      sessionStorage.setItem(COMPARISON_STORAGE_KEY, JSON.stringify(set));
    }

    export function addToComparison(profile: CreatorProfile): void {
      const id = `${profile.platform}:${profile.profileUrl}`;
      const set = getComparisonSet();
      if (!set.creatorIds.includes(id) && set.creatorIds.length < 5) {
        set.creatorIds.push(id);
        setComparisonSet(set);
      }
    }

    export function removeFromComparison(profile: CreatorProfile): void {
      const id = `${profile.platform}:${profile.profileUrl}`;
      const set = getComparisonSet();
      set.creatorIds = set.creatorIds.filter((c) => c !== id);
      setComparisonSet(set);
    }

    export function clearComparison(): void {
      setComparisonSet({ creatorIds: [] });
    }
    ```
