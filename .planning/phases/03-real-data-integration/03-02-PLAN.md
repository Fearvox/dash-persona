    **Component 1: `src/components/ui/collected-at.tsx`** (per D-04, D-05, D-06)

    Create a `'use client'` component that renders a `collectedAt` timestamp as a relative-time badge.

    ```typescript
    'use client';

    import { formatRelativeTime, isStale } from '@/lib/utils/relative-time';
    import { getLocale } from '@/lib/i18n';

    interface CollectedAtProps {
      timestamp: string;  // ISO-8601
      className?: string;
    }

    export function CollectedAt({ timestamp, className }: CollectedAtProps) {
      const locale = getLocale();
      const relative = formatRelativeTime(timestamp, locale);
      const stale = isStale(timestamp, 7);

      return (
        <time
          dateTime={timestamp}
          title={new Date(timestamp).toLocaleString()}
          className={`font-mono text-[0.6875rem] tabular-nums ${
            stale ? 'text-[var(--accent-yellow)]' : 'text-[var(--text-subtle)]'
          } ${className ?? ''}`}
        >
          {relative}
        </time>
      );
    }
    ```

    Key details:
    - Font: Geist Mono via `font-mono` class (per D-04)
    - Color: `--text-subtle` default, `--accent-yellow` when stale >7 days (per D-05)
    - Hover: `title` attribute shows full ISO timestamp as native tooltip (per D-04)
    - Size: `text-[0.6875rem]` (11px) -- subtle, doesn't compete with metrics
    - `tabular-nums` for consistent digit width
    - No emoji, no inline styles
